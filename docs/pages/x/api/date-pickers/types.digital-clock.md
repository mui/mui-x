type DigitalClockProps = {
  /**
   * Available views.
   * @default ['hours']
   */
  views?: 'hours'[];
  /** Override or extend the styles applied to the component. */
  classes?: Partial<DigitalClockClasses>;
  /**
   * Overrideable component slots.
   * @default {}
   */
  slots?: DigitalClockSlots;
  /**
   * The props used for each component slot.
   * @default {}
   */
  slotProps?: DigitalClockSlotProps;
  /**
   * If `true`, disable values before the current date for date components, time for time components and both for date time components.
   * @default false
   */
  disablePast?: boolean;
  /**
   * If `true`, disable values after the current date for date components, time for time components and both for date time components.
   * @default false
   */
  disableFuture?: boolean;
  /**
   * Minimal selectable time.
   * The date part of the object will be ignored unless `props.disableIgnoringDatePartForTimeValidation === true`.
   */
  minTime?: any;
  /**
   * Maximal selectable time.
   * The date part of the object will be ignored unless `props.disableIgnoringDatePartForTimeValidation === true`.
   */
  maxTime?: any;
  /**
   * Step over minutes.
   * @default 1
   */
  minutesStep?: number;
  /**
   * Disable specific time.
   * @returns If `true` the time will be disabled.
   */
  shouldDisableTime?: (value: any, view: TimeView) => boolean;
  /**
   * Do not ignore date part when validating min/max time.
   * @default false
   */
  disableIgnoringDatePartForTimeValidation?: boolean;
  /**
   * Choose which timezone to use for the value.
   * Example: "default", "system", "UTC", "America/New_York".
   * If you pass values from other timezones to some props, they will be converted to this timezone before being used.
   * @default The timezone of the `value` or `defaultValue` prop is defined, 'default' otherwise.
   * @see
   */
  timezone?: PickersTimezone;
  /**
   * 12h/24h view for hour selection clock.
   * @default adapter.is12HourCycleInCurrentLocale()
   */
  ampm?: boolean;
  /**
   * The time steps between two time options.
   * For example, if `timeStep = 45`, then the available time options will be `[00:00, 00:45, 01:30, 02:15, 03:00, etc.]`.
   * @default 30
   */
  timeStep?: number;
  /**
   * If `true`, disabled digital clock items will not be rendered.
   * @default false
   */
  skipDisabled?: boolean;
  className?: string;
  /** The system prop that allows defining system overrides as well as additional CSS styles. */
  sx?: SxProps<Theme>;
  /**
   * The selected value.
   * Used when the component is controlled.
   */
  value?: any | null;
  /**
   * The default selected value.
   * Used when the component is not controlled.
   */
  defaultValue?: any | null;
  /**
   * The date used to generate the new value when both `value` and `defaultValue` are empty.
   * @default The closest valid time using the validation props, except callbacks such as `shouldDisableTime`.
   */
  referenceDate?: any;
  /**
   * Callback fired when the value changes.
   * @template The value type. It will be the same type as `value` or `null`. It can be in `[start, end]` format in case of range value.
   * @template The view type. Will be one of date or time views.
   */
  onChange?: (value: any, selectionState?: PickerSelectionState, selectedView?: 'hours') => void;
  /**
   * The default visible view.
   * Used when the component view is not controlled.
   * Must be a valid option from `views` list.
   */
  openTo?: 'hours';
  /**
   * Callback fired on view change.
   * @template Type of the view. It will vary based on the Picker type and the `views` it uses.
   */
  onViewChange?: (view: 'hours') => void;
  /**
   * The visible view.
   * Used when the component view is controlled.
   * Must be a valid option from `views` list.
   */
  view?: 'hours';
  /**
   * If `true`, the main element is focused during the first mount.
   * This main element is:
   * - the element chosen by the visible view if any (i.e: the selected day on the `day` view).
   * - the `input` element if there is a field rendered.
   */
  autoFocus?: boolean;
  /** Controlled focused view. */
  focusedView?: 'hours' | null;
  /**
   * Callback fired on focused view change.
   * @template Type of the view. It will vary based on the Picker type and the `views` it uses.
   */
  onFocusedViewChange?: (view: 'hours', hasFocus: boolean) => void;
  /**
   * If `true`, the component is disabled.
   * When disabled, the value cannot be changed and no interaction is possible.
   * @default false
   */
  disabled?: boolean;
  /**
   * If `true`, the component is read-only.
   * When read-only, the value cannot be changed but the user can interact with the interface.
   * @default false
   */
  readOnly?: boolean;
};
